/* ── HELPERS ── */
function fmt(n,d=4){ return Number(n).toFixed(d); }
function nextStake(c){ return c.stakers.length ? c.stakers[c.stakers.length-1].staked*(1+c.spread/100) : 0.01; }
function entryRequired(c){ return nextStake(c)*(1+FEE); }
function nextProfit(c){ return nextStake(c)*(c.spread/100); }
function donPct(c){ return c.stakers.length ? Math.min((c.donPot/c.stakers[c.stakers.length-1].staked)*100,100):0; }
function isMeLast(c){ return c.stakers.length>0 && c.stakers[c.stakers.length-1].me; }
function hasMe(c){ return c.stakers.some(s=>s.me); }
function myYieldShare(c){
  if(!c.stakers.length || !c.totalStaked) return 0;
  const myTotal = c.stakers.filter(s=>s.me).reduce((a,s)=>a+s.staked,0);
  return c.yieldPool * (myTotal/c.totalStaked);
}
function timerExpired(c){
  if(!c.stakers.length) return true;
  const last = c.stakers[c.stakers.length-1];
  if(!last.enteredAt) return true;
  return Date.now() - last.enteredAt >= COOLDOWN_MS;
}
function timerRemaining(c){
  if(!c.stakers.length) return 0;
  const last = c.stakers[c.stakers.length-1];
  if(!last.enteredAt) return 0;
  return Math.max(0, COOLDOWN_MS - (Date.now()-last.enteredAt));
}
function fmtTimer(ms){
  if(ms<=0) return 'Expired';
  const m=Math.floor(ms/60000), s=Math.floor((ms%60000)/1000);
  return `${m}:${String(s).padStart(2,'0')}`;
}

let timerInterval=null;
function startTimerTick(){
  if(timerInterval) clearInterval(timerInterval);
  timerInterval=setInterval(()=>{
    const el=document.getElementById('timerVal');
    if(!el) return;
    const chain=CHAINS[currentChainId];
    const expired=timerExpired(chain);
    const remaining=timerRemaining(chain);
    el.textContent=expired?'Expired — Break available':fmtTimer(remaining);
    el.className='timer-val'+(expired?' expired':'');
    if(expired) clearInterval(timerInterval);
  },1000);
}

/* ── ACTIONS ── */
function doEnter(chainId){
  if(!walletConnected){ toast('Connect wallet first','err'); return; }
  const chain=CHAINS[chainId];
  const ns=nextStake(chain),er=entryRequired(chain),np=nextProfit(chain);
  chain.stakers.forEach(s=>s.last=false);
  chain.stakers.push({addr:ME,staked:ns,profit:np,last:true,me:true,enteredAt:Date.now()});
  chain.totalStaked+=ns;
  chain.status=chain.stakers.length>=5?'hot':'active';
  myPositions.forEach(p=>{if(p.chainId===chainId) p.isLast=false;});
  myPositions.push({chainId,staked:ns,profit:np,isLast:true,posId:`p${Date.now()}`,enteredAt:Date.now()});
  closeModal(); renderChainPage(); buildChainDropdown();
  toast(`Entered ${chain.label} chain — paid ${fmt(er)} SOL`,'ok');
}

function submitDonate(chainId){
  const input=document.getElementById('dAmt');
  const amt=input?parseFloat(input.value):0;
  if(!amt||amt<=0){ toast('Enter a valid amount','err'); return; }
  if(!walletConnected){ toast('Connect wallet first','err'); return; }
  const chain=CHAINS[chainId];
  const lastS=chain.stakers[chain.stakers.length-1];
  const target=lastS?lastS.staked:0.01;
  chain.donPot=parseFloat((chain.donPot+amt).toFixed(8));
  const forced=chain.donPot>=target-0.000001;
  const yield_amt=chain.yieldPool;
  const pot_amt=chain.donPot;
  closeModal();
  if(forced){
    chain.stakers=[];chain.yieldPool=0;chain.donPot=0;chain.status='empty';chain.totalStaked=0;
    myPositions=myPositions.filter(p=>p.chainId!==chainId);
    renderChainPage(); renderProfile(); buildChainDropdown();
    toast(`${chain.label} chain auto-broke! Community splits ${fmt(yield_amt)} SOL yield pool. Last holder got ${fmt(pot_amt)} SOL pot.`,'ok');
  } else {
    renderChainPage(); renderProfile();
    toast(`Donated ${fmt(amt)} SOL to ${chain.label} bounty pot`,'ok');
  }
}

function doBreak(chainId){
  if(!timerExpired(CHAINS[chainId])){ toast('Timer not expired yet','err'); return; }
  const chain=CHAINS[chainId];
  const y=chain.yieldPool;
  chain.stakers=[];chain.yieldPool=0;chain.donPot=0;chain.status='empty';chain.totalStaked=0;
  myPositions=myPositions.filter(p=>p.chainId!==chainId);
  closeModal(); renderChainPage(); renderProfile(); buildChainDropdown();
  toast(`Chain broken! You collected ${fmt(y)} SOL yield pool 🎉`,'ok');
}

function toast(msg,type='ok'){
  const c=document.getElementById('toasts');
  const t=document.createElement('div');
  t.className=`toast ${type}`;
  t.innerHTML=`<span>${type==='ok'?'✓':'✕'}</span> <span>${msg}</span>`;
  c.appendChild(t);
  setTimeout(()=>t.remove(),4500);
}
